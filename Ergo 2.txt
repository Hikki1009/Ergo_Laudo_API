from flask import Flask, request, send_file, jsonify
from flask_cors import CORS
import json
import io
import traceback
from gerar_pdf import gerar_pdf

app = Flask(__name__)
CORS(app)

@app.route("/", methods=["GET"])
def health():
    return jsonify({"status": "ok", "message": "ErgoLaudo API funcionando"})

@app.route("/gerar-pdf", methods=["POST"])
def gerar():
    try:
        laudo = request.get_json()
        if not laudo:
            return jsonify({"erro": "JSON inválido"}), 400

        buffer = io.BytesIO()
        gerar_pdf(laudo, buffer)
        buffer.seek(0)

        empresa = laudo.get("dadosGerais", {}).get("empresa", {}).get("razaoSocial", "laudo")
        nome_arquivo = f"AET_{empresa[:30].replace(' ', '_')}.pdf"

        return send_file(
            buffer,
            mimetype="application/pdf",
            as_attachment=True,
            download_name=nome_arquivo
        )
    except Exception as e:
        traceback.print_exc()
        return jsonify({"erro": str(e)}), 500

if __name__ == "__main__":
    app.run(debug=True)
